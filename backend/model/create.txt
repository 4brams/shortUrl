  id SERIAL PRIMARY KEY,

CREATE TABLE account (
  id SERIAL PRIMARY KEY,
  account text NOT NULL,
  password text NOT NULL,
  time TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
);
